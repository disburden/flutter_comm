static Widget splitLine({
	double height = 8.0,
	double leftIndent = 0.0,
	double rightIndent = 0.0,
	Color lineColor = const Color(0xfff7f7f7)
}) {
	return Padding(
		padding: EdgeInsets.only(left: leftIndent, right: rightIndent),
		child: Container(
			color: lineColor,
			height: height,
		),
	);
}
